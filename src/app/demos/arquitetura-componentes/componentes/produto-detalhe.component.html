// Reage quando o ID muda
// sem recriar o componente
this.route.paramMap.subscribe(params =>
  const id = params.get('id');
);
⚠️ snapshot (leitura única) 
// Funciona apenas na primeira
// carga do componente
const id =
  this.route.snapshot
    .paramMap.get('id');
